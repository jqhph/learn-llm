def extract_order_no(query: str) -> str|None:
    for pat in [r'SK\d{6,}', r'ORD\d{6,}', r'\b\d{10,}\b']:
        if m := re.search(pat, query):
            return m.group()
    return llm_extract(query)  # 兜底

四、生产架构部署检查清单

检查项要求
向量库数据范围只含知识型/参考型数据，不含事务型数据
订单查询方式API / Function Calling 直连订单服务，不经过向量库
用户隐私姓名、电话、地址不出现在持久化存储的向量库中
NL2SQL 安全防护只读用户 + 参数化查询 + LIMIT + 超时 + 敏感列脱敏
路由兜底无法分类时走人工或通用 FAQ 搜索，不返回空结果
正则 + LLM 分层结构化提取（订单号/SKU）先正则、后 LLM，不只用 LLM
审计日志每次订单/用户数据查询记录在案，可追溯
