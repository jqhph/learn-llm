三、服务器初始化命令速查

# Docker 安装（阿里云镜像）
curl -fsSL https://get.docker.com | bash -s docker --mirror Aliyun

# Docker 镜像加速配置
sudo tee /etc/docker/daemon.json <<'EOF'
{"registry-mirrors":["https://<你的地址>.mirror.aliyuncs.com"],
 "log-driver":"json-file",
 "log-opts":{"max-size":"10m","max-file":"3"},
 "exec-opts":["native.cgroupdriver=systemd"]}
EOF
sudo systemctl daemon-reload && sudo systemctl restart docker

# UFW 防火墙
ufw allow 22/tcp && ufw allow 80/tcp && ufw allow 443/tcp && ufw --force enable

# 系统优化
echo "vm.swappiness=10" >> /etc/sysctl.conf && sysctl -p
timedatectl set-timezone Asia/Shanghai

四、Let's Encrypt SSL 证书

# 安装 Certbot
sudo apt install -y certbot python3-certbot-nginx

# 申请证书（需先确保 80 端口可访问）
sudo certbot --nginx -d your-domain.com

# 自动续期（systemd timer）
sudo tee /etc/systemd/system/certbot-renew.service <<'SVC'
[Unit]
Description=Certbot Renewal
After=network-online.target
[Service]
Type=oneshot
ExecStart=/usr/bin/certbot renew --post-hook "docker exec rag-nginx nginx -s reload"
SVC

sudo tee /etc/systemd/system/certbot-renew.timer <<'TMR'
[Unit]
Description=Certbot renewal twice daily
[Timer]
OnCalendar=*-*-* 00,12:00:00
RandomizedDelaySec=1h
Persistent=true
[Install]
WantedBy=timers.target
TMR
sudo systemctl enable --now certbot-renew.timer

# 测试续期
sudo certbot renew --dry-run

五、Nginx HTTPS 关键配置片段

# HTTP → HTTPS 重定向
server {
    listen 80;
    server_name your-domain.com;
    return 301 https://$server_name$request_uri;
}

# HTTPS 服务
server {
    listen 443 ssl http2;
    server_name your-domain.com;

    ssl_certificate /etc/letsencrypt/live/your-domain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/your-domain.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_session_cache shared:SSL:10m;

    add_header Strict-Transport-Security "max-age=63072000" always;
    add_header X-Content-Type-Options nosniff;
    add_header X-Frame-Options DENY;

    client_max_body_size 50m;

    location /api/ {
        rewrite ^/api(/.*)$ $1 break;
        proxy_pass http://rag_api;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 120s;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }

    # 生产环境关闭 API 文档
    location /docs { deny all; return 404; }
    location /redoc { deny all; return 404; }
}

六、GitHub Actions CI/CD 核心 Workflow

# .github/workflows/deploy.yml
name: Deploy RAG Service
on:
  push:
    branches: [main]
  workflow_dispatch:

env:
  REGISTRY: registry.cn-hangzhou.aliyuncs.com
  NAMESPACE: your-ns
  IMAGE_NAME: rag-service

jobs:
  build-and-push:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: '3.12', cache: 'pip' }
      - run: pip install -r requirements.txt pytest
      - run: pytest tests/ -v --timeout=30
        env: { OPENAI_API_KEY: ${{ secrets.OPENAI_API_KEY }} }
      - uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ secrets.ACR_USERNAME }}
          password: ${{ secrets.ACR_PASSWORD }}
      - uses: docker/build-push-action@v6
        with:
          context: . ; file: ./Dockerfile
          platforms: linux/amd64
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.NAMESPACE }}/${{ env.IMAGE_NAME }}:latest
            ${{ env.REGISTRY }}/${{ env.NAMESPACE }}/${{ env.IMAGE_NAME }}:${{ github.sha }}

  deploy:
    runs-on: ubuntu-latest
    needs: build-and-push
    steps:
      - uses: appleboy/ssh-action@v1.0.3
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: ${{ secrets.SERVER_USERNAME }}
          key: ${{ secrets.SERVER_SSH_KEY }}
          script: |
            cd /opt/rag-service
            git pull origin main
            docker login ${{ env.REGISTRY }} \
              -u ${{ secrets.ACR_USERNAME }} \
              -p ${{ secrets.ACR_PASSWORD }}
            docker pull ${{ env.REGISTRY }}/${{ env.NAMESPACE }}/${{ env.IMAGE_NAME }}:latest
            export IMAGE_TAG=${{ env.REGISTRY }}/${{ env.NAMESPACE }}/${{ env.IMAGE_NAME }}:latest
            docker compose down --remove-orphans || true
            docker compose up -d
            sleep 15
            curl -f http://localhost:8000/health || exit 1

九、蓝绿部署核心逻辑

# 核心思想
# 同时运行两套 RAG API（blue + green）
# Nginx 指向当前活跃环境
# 部署时更新非活跃环境 → 验证 → 切换 Nginx

# 关键切换命令
sed -i 's|proxy_pass http://rag_api_blue|proxy_pass http://rag_api_green|' nginx.conf
docker exec rag-nginx nginx -s reload

# 快速回滚：把 proxy_pass 改回蓝环境，reload Nginx

十、上线安全检查清单

  • ☐ 安全组：只放行 22/80/443，8000/6379 不暴露
  • ☐ SSH：禁用密码登录（PasswordAuthentication no）
  • ☐ HTTPS：HTTP → HTTPS 强制 301 重定向
  • ☐ API 文档：生产环境关闭 /docs /redoc
  • ☐ .env 密钥：不提交到 Git，.gitignore 中包含 .env
  • ☐ Docker 日志：daemon.json 配置了 max-size / max-file
  • ☐ SSL 续期：systemd timer 已启用，certbot renew --dry-run 成功
  • ☐ 备份：Chroma 数据已持久化，数据目录已备份
  • ☐ 自愈：测试 docker stop 后自动重启，测试 sudo reboot 后服务恢复
  • ☐ 回滚：知道如何回滚到上一个版本
